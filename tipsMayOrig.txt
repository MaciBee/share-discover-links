<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Submit Your Tip</title>
</head>
<body>
    <h1>Submit a Tip</h1>
    <form id="tipForm">
        <input type="text" name="title" placeholder="Title" required><br>
        <textarea name="description" placeholder="Description" required></textarea><br>
        <input type="text" name="category" placeholder="Category"><br>
        <input type="text" name="author" placeholder="Author"><br>
        <button type="submit">Submit</button>
    </form>
    <div id="tipsContainer"></div>

    <script>
    document.addEventListener('DOMContentLoaded', function() {
        const form = document.getElementById('tipForm');
        form.addEventListener('submit', function(event) {
            event.preventDefault();
            const formData = new FormData(form);
            fetch('http://autodidacting.org:3001/tips', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    title: formData.get('title'),
                    description: formData.get('description'),
                    category: formData.get('category'),
                    author: formData.get('author')
                })
            })
            .then(response => response.json())
            .then(data => {
                const container = document.getElementById('tipsContainer');
                const div = document.createElement('div');
                div.innerHTML = `<h4>${data.title}</h4><p>${data.description}</p><p>Category: ${data.category}</p><p>Author: ${data.author}</p>`;
                container.prepend(div); // Adds the new tip at the top
                form.reset();
            })
            .catch(error => {
                console.error('Error submitting tip:', error);
                alert('Failed to submit tip.');
            });
        });

        // Fetch and display existing tips on load
        fetch('http://autodidacting.org:3001/tips')
        .then(response => response.json())
        .then(tips => {
            const container = document.getElementById('tipsContainer');
            tips.forEach(tip => {
                const div = document.createElement('div');
                div.innerHTML = `<h4>${tip.title}</h4><p>${tip.description}</p><p>Category: ${tip.category}</p><p>Author: ${tip.author}</p>`;
                container.appendChild(div);
            });
        })
        .catch(error => console.error('Error fetching tips:', error));
    });
    </script>
</body>
</html>
