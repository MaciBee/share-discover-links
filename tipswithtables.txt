    <h2>Welcome to Tips</h2>

<p>Think of them as throw away tweets. Here you can capture fleeting thoughts</p>
	<span id="showFormLink">Click here to submit </span> 
    <form id="tipForm">
        <input type="text" name="title" placeholder="Title" required><br>
        <textarea name="description" placeholder="Description" required></textarea><br>
        <input type="text" name="category" placeholder="Category"><br>
        <input type="text" name="author" placeholder="Author"><br>
        <button type="submit">Submit</button>
    </form>
    <div id="tipsContainer"></div>

    <script>
    document.addEventListener('DOMContentLoaded', function() {
        const formLink = document.getElementById('showFormLink');
        const form = document.getElementById('tipForm');

        formLink.addEventListener('click', function() {
            form.style.display = form.style.display === 'block' ? 'none' : 'block'; // Toggle form display
        });

        form.addEventListener('submit', function(event) {
            event.preventDefault();
            const formData = new FormData(form);
            fetch('http://autodidacting.org:3001/tips', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    title: formData.get('title'),
                    description: formData.get('description'),
                    category: formData.get('category'),
                    author: formData.get('author')
                })
            })
            .then(response => response.json())
            .then(data => {
                const container = document.getElementById('tipsContainer');
                const div = document.createElement('div');
                div.className = 'card';
                div.innerHTML = `<h4>${data.title}</h4><p>${data.description}</p><div class='details'><div class='detail-item'>Category: ${data.category}</div><div class='detail-item'>Author: ${data.author}</div></div>`;
                container.prepend(div); // Adds the new tip at the top
                form.style.display = 'none'; // Hide form after submission
                form.reset(); // Reset form fields after successful submission
            })
            .catch(error => {
                console.error('Error submitting tip:', error);
                alert('Failed to submit tip.');
            });
        });

        // Fetch and display existing tips on load
        fetch('http://autodidacting.org:3001/tips')
        .then(response => response.json())
        .then(tips => {
            const container = document.getElementById('tipsContainer');
            tips.forEach(tip => {
                const div = document.createElement('div');
                div.className = 'card';
                div.innerHTML = `<h4>${tip.title}</h4><p>${tip.description}</p><div class='details'><div class='detail-item'>Category: ${tip.category}</div><div class='detail-item'>Author: ${tip.author}</div></div>`;
                container.appendChild(div);
            });
        })
        .catch(error => console.error('Error fetching tips:', error));
    });
    </script>
